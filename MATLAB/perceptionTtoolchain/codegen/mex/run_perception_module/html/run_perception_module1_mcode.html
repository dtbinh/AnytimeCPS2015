1function run_perception_module(inImgName)
 2
 3%folder = '../../Data/Barrel';
 4%images = dir([folder,'/*.png']);
 5GMs = load('GMs.mat');
 6
 7colors = [1,2,3];
 8nbcolors = length(colors);
 9poiGM = GMs.poiGM;
10nonpoiGM = GMs.nonpoiGM;
11minAcceptanceProbScalingFactor = 0.8;
12minSignificanceProb = 0.2;
13SVMModel = GMs.SVMModel;
14
15
16% pixels of interest (poi)
17poi = [];
18nonpoi = [];
19
20% Cluster the data
21minAcceptanceProb = minAcceptanceProb/minAcceptanceProbScalingFactor; % allow for 1st iteration of while loop
22Im=imread(inImgName);
23I = preprocess_img(Im);
24I = I(:,:,colors);
25a=size(I);
26candidate = reshape(I,a(1)*a(2),nbcolors);
27[idxpoi, ~, posteriorspoi] = cluster(poiGM,double(candidate));
28[idxnon, ~, posteriorsnon] = cluster(nonpoiGM,double(candidate));
29b = size(candidate);
30% For each pix, decide if it's best described by poi or non
31d1 = sum(posteriorspoi.*repmat(poiGM.ComponentProportion,b(1),1),2);
32d2 = sum(posteriorsnon.*repmat(nonpoiGM.ComponentProportion,b(1),1),2);
33clusterObjOfInterest = 0; %initialize while
34while (nnz(clusterObjOfInterest)==0 && minAcceptanceProb >= minSignificanceProb)
35    minAcceptanceProb = minAcceptanceProb*minAcceptanceProbScalingFactor;
36    clusterObjOfInterest = (d1 > max(d2,minAcceptanceProb));
37end
38% Filter out the smaller things
39M = imerode(reshape(clusterObjOfInterest,a(1),a(2)),strel('rectangle',[1,1]));
40M = medfilt2(M,'symmetric');
41% Connect components and extract features for the positive class, i.e. of barrel objects.
42statsPosClass = regionprops(M, 'MajorAxisLength', 'MinorAxisLength', 'Eccentricity', 'Solidity', 'Centroid', 'Boundingbox');
43nbFoundObjects = length(statsPosClass);
44if ~isempty(statsPosClass)
45    % Vector = major, minor, ecc, solidity
46    featuresPosClass = zeros(nbFoundObjects, 3);
47    for i=1:nbFoundObjects
48        featuresPosClass(i,:) = [statsPosClass(i).MajorAxisLength/statsPosClass(i).MinorAxisLength, statsPosClass(i).Eccentricity, statsPosClass(i).Solidity];
49    end
50    [label,score]=predict(SVMModel, featuresPosClass);
51    % Discard low-scoring objects
52    % ..??
53    selectedObj = statsPosClass(label==1); % 1 is the positive class index
54    %         figure
55    %         imshow(Im(:,:,1));
56    %         hold on
57    %         centroids = cat(1, selectedObj.Centroid);
58    %         plot(centroids(:,1), centroids(:,2), 'b*')
59end
60
61
62
63
64end
