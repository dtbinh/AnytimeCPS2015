1function [phi,psi] = MeyerWavelet(t)
 2%% Meyer wavelet (psi) and scale (phi) as in Valenzuela and Oliviera
 3a1 = (2*pi)/3;
 4a2 = 4/3;
 5a3 = 4*pi/3;
 6a4 = pi;
 7a5 = (16*pi)/9;
 8
 9
10if(t==0 || t==sqrt(a4/a5)|| t==-sqrt(a4/a5))
11    %phi = 2/3 + 4/(3*pi);
12    phi = (a1*cos(a1*t) + a2*cos(a3*t) - a2*a3*t*sin(a3*t))/ ...
13        (a4-3*a5*t^2);
14else
15    phi = (sin((2*pi/3)*t) + (4*t/3)*cos(4*pi*t/3))/ ...
16        (pi*t-(16*pi*t^3)/9);
17end
18
19%evalue psi_1
20z=t-0.5;
21
22c1 = 4/(3*pi);
23c2 = 2*pi/3;
24c3 = (1/pi);
25c4 = 4*pi/3;
26c5 = 16/9;
27
28k1 = 8/(3*pi);
29k2 = 8*pi/3;
30k3 = 1/pi;
31k4 = 4*pi/3;
32k5 = 64/9;
33
34%L'HopyBoy's rule at the points of 0/0 for psi_1
35if(z==0 || z== 3/4 || z==-3/4)
36    psi_1_num = -c1*c2*z*sin(c2*z)+c1*cos(c2*z)-c4*c3*cos(c4*z);
37    psi_1_den = 1-3*c5*z^2;
38    psi_1 = psi_1_num/psi_1_den;
39    [z psi_1];
40else
41    psi_1_num = (4/(3*pi))*(t-0.5)*cos((2*pi/3)*(t-0.5))- ...
42        (1/pi)*sin((4*pi/3)*(t-0.5));
43    psi_1_den = (t-0.5)-(16/9)*(t-0.5)^3;
44    psi_1 = psi_1_num/psi_1_den;
45end
46% for psi_2 now
47if(z==0 || z==3/8 || z==-3/8)
48    psi_2_num = -k1*k2*z*sin(k2*z)+k1*cos(k2*z)+k4*k3*cos(k4*z);
49    psi_2_den = 1-3*k5*z^2;
50    psi_2 = psi_2_num/psi_2_den;
51    [z psi_2];
52else
53    psi_2_num = (8/(3*pi))*(t-0.5)*cos((8*pi/3)*(t-0.5))+ ...
54        (1/pi)*sin((4*pi/3)*(t-0.5));
55    psi_2_den = (t-0.5)-(64/9)*(t-0.5)^3;
56    psi_2 = psi_2_num/psi_2_den;
57end
58
59psi = psi_1 + psi_2;
